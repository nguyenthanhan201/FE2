    <script>
        document.getElementById("drawChart").addEventListener("click", () => {
            const data = document.getElementById("data").value
            arrayData = data.split(',')
            console.log(arrayData)
            const toNumbers = arr => arr.map(Number);
            arrayData = toNumbers(arrayData);
            console.log(arrayData.length)
            if (arrayData.length > 8 || arrayData.length == 0) {
                alert('Phai co du lieu va gioi han 8 gia tri')
                return
            }
            //tim max
            const max = Math.max(...arrayData)
            console.log(max)
            const arrayImg = document.getElementById("arrayImg")
            arrayImg.textContent = ""
            arrayData.map((val) => {
                const child = document.createElement("div")
                child.textContent = val
                child.style = "width: 75px; border-radius: 10px;"
                if (val == max)
                    child.style.height = `500px`
                else
                    child.style.height = `${500 * val / max}px`
                child.classList = "bg-warning text-center  mr-3"
                arrayImg.appendChild(child);
            })
        })
    </script>
